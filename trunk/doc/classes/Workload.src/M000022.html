    def linkUsers
        size=@users.size()
        @tasks.each{|task|
            thisUserID=(rand(size.to_i))
            puts "considering task #{task}, userID = #{thisUserID}"
            task.eachJob{|job|
                puts "subjob: userID = #{thisUserID}"
                job.userID=thisUserID
            }
        }
    end
