    def parseSWF(swf)
        @swf=swf    
        swf.each {|line|
            if line !~ /^;/         #this is not a comment line
                current=AtomicJob.new
                current.readSWFFormat(line)
                #print "#{current.jobID} = "+current.to_s
                @jobs[current.jobID.to_i]=current
            end
        }
        @jobs.compact!               # remove all nil elements
    end
