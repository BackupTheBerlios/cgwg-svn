    def initialize
        @workloads=Hash.new
        # Now, we build the indices we need to have in the workload. 
        # We have for each interval @@config.loadDeviation a slot.
        0.step(1, @@config.loadDeviation) {|step|
            @workloads[step]=nil
        }
    end
